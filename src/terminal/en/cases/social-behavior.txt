# Data loading and preview

import pandas as pd
data = pd.read_csv('data_behav/IFD-Data500-Final.csv')
print(f"Data dimensions: {data.shape}")
print(f"Variable list: {list(data.columns)}")
data.head()

# Exploratory factor analysis
from factor_analyzer import FactorAnalyzer
import numpy as np

# Select analysis variables
variables = ['BI1', 'BI2', 'BI3', 'PU1', 'PU2', 'PU3', 
            'PEU1', 'PEU2', 'PEU3', 'ATT1', 'ATT2', 'ATT3',
            'IFDA1', 'IFDA2', 'IFDA3', 'BURE1', 'BURE2']

# Perform factor analysis
fa = FactorAnalyzer(n_factors=6, rotation='varimax')
fa.fit(data[variables])

# Get analysis results
loadings = fa.loadings_
communalities = fa.get_communalities()
variance_explained = fa.get_factor_variance()

print("Factor loading matrix:")
print(loadings)
print(f"\nCumulative explained variance: {variance_explained[2]:.1%}")
